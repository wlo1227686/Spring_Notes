DROP TABLE IF EXISTS employees;

CREATE TABLE employees(
ID INT(7) NOT NULL Auto_Increment Primary Key,
LAST_NAME VARCHAR(10),
EMAIL VARCHAR(100),
DEPT_ID INT(7)
)CHARACTER SET utf8 COLLATE utf8_general_ci;
