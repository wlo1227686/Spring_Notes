DROP TABLE IF EXISTS book;

CREATE TABLE book(
ISBN VARCHAR(10),
BOOK_NAME VARCHAR(10),
PRICE INT(7)
)CHARACTER SET utf8 COLLATE utf8_general_ci;

INSERT INTO book VALUE(1001,'Java',100);

INSERT INTO book VALUE(1002,'Oracle',70);


DROP TABLE IF EXISTS account;

CREATE TABLE account(
USERNAME VARCHAR(10),
BALANCE INT(7)
)CHARACTER SET utf8 COLLATE utf8_general_ci;

INSERT INTO account VALUE('AA',200);

DROP TABLE IF EXISTS book_stock;

CREATE TABLE book_stock(
ISBN VARCHAR(10),
STOCK INT(7)
)CHARACTER SET utf8 COLLATE utf8_general_ci;

INSERT INTO book_stock VALUE(1001,4);

INSERT INTO book_stock VALUE(1002,8);
